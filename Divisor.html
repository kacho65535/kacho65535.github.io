//nの約数を小さい順に列挙した配列を返す
//計算量:O(√n)
vector<long long> divisor(long long n)
{
    vector<long long> ret1, ret2;
    for (long long i = 1; i * i <= n; i++)
    {
        if (n % i == 0)
        {
            ret1.push_back(i);
            if (i * i != n)
                ret2.push_back(n / i);
        }
    }
    long long siz = ret2.size();
    for (long long i = (siz - 1); i >= 0; i--)
    {
        ret1.push_back(ret2[i]);
    }
    return (ret1);
}
