template <typename T>
bool primejudge(T n)
{
  if (n < 2)
    return false;
  else if (n == 2)
    return true;
  else if (n % 2 == 0)
    return false;
  double sqrtn = sqrt(n);
  for (T i = 3; i < sqrtn + 1; i++)
  {
    if (n % i == 0)
    {
      return false;
    }
    i++;
  }
  return true;
}
